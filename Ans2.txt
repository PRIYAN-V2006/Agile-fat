---

LeaveService.java

package com.example;

public class LeaveService {

    public String applyLeave(Employee emp, LeaveRequest req) {
        if (req.days <= emp.leaveBalance) {
            emp.leaveBalance -= req.days;
            req.status = "APPROVED";
        } else {
            req.status = "REJECTED";
        }
        return req.status;
    }
}

Step 3: JUnit Test Cases

package com.example;

import org.junit.Test;
import static org.junit.Assert.*;

public class LeaveServiceTest {

    @Test
    public void testApproved() {
        Employee emp = new Employee("A");
        LeaveRequest req = new LeaveRequest(5);
        LeaveService service = new LeaveService();

        assertEquals("APPROVED", service.applyLeave(emp, req));
    }

    @Test
    public void testRejected() {
        Employee emp = new Employee("A");
        LeaveRequest req = new LeaveRequest(25);
        LeaveService service = new LeaveService();

        assertEquals("REJECTED", service.applyLeave(emp, req));
    }
}

Step 5: CI/CD Pipeline (GitHub Actions)

Create file:

name: CI Pipeline

on:
  push:
    branches: [ "main" ]

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
    - uses: actions/checkout@v3

    - name: Setup Java
      uses: actions/setup-java@v3
      with:
        java-version: '17'

    - name: Build with Maven
      run: mvn clean install
