Employee Leave Management System

1. Leave Request and Approval Logic (Java)

class Employee {
    String name;
    int leaveBalance = 20;

    Employee(String name) {
        this.name = name;
    }
}

class LeaveRequest {
    int days;
    String status;

    LeaveRequest(int days) {
        this.days = days;
        this.status = "PENDING";
    }
}

class LeaveService {
    public String applyLeave(Employee emp, LeaveRequest req) {
        if (req.days <= emp.leaveBalance) {
            req.status = "APPROVED";
            emp.leaveBalance -= req.days;
        } else {
            req.status = "REJECTED";
        }
        return req.status;
    }
}

2. JUnit Test Cases

import static org.junit.Assert.*;
import org.junit.Test;

public class LeaveServiceTest {

    @Test
    public void testLeaveApproved() {
        Employee emp = new Employee("John");
        LeaveRequest req = new LeaveRequest(5);
        LeaveService service = new LeaveService();

        assertEquals("APPROVED", service.applyLeave(emp, req));
    }

    @Test
    public void testLeaveRejected() {
        Employee emp = new Employee("John");
        LeaveRequest req = new LeaveRequest(25);
        LeaveService service = new LeaveService();

        assertEquals("REJECTED", service.applyLeave(emp, req));
    }
}
